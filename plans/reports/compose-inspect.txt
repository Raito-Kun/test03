==== crm compose prod override (if exists) ====
$ cat /opt/crm/docker-compose.prod.yml 2>/dev/null || echo '(no prod override)'
services:
  postgres:
    image: postgres:15-alpine
    container_name: crm-postgres
    restart: unless-stopped
    environment:
      POSTGRES_USER: ${DB_USER:-crm_user}
      POSTGRES_PASSWORD: ${DB_PASSWORD:-crm_password}
      POSTGRES_DB: ${DB_NAME:-crm_db}
    volumes:
      - pgdata:/var/lib/postgresql/data
    networks:
      - internal
    healthcheck:
      test: ['CMD-SHELL', 'pg_isready -U ${DB_USER:-crm_user} -d ${DB_NAME:-crm_db}']
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    container_name: crm-redis
    restart: unless-stopped
    volumes:
      - redisdata:/data
    networks:
      - internal
    healthcheck:
      test: ['CMD', 'redis-cli', 'ping']
      interval: 10s
      timeout: 5s
      retries: 5

  backend:
    build:
      context: .
      dockerfile: packages/backend/Dockerfile
    container_name: crm-backend
    restart: unless-stopped
    env_file: .env
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    networks:
      - internal
      - external
    healthcheck:
      test: ['CMD', 'wget', '-qO-', 'http://localhost:4000/api/v1/health']
      interval: 30s
      timeout: 10s
      retries: 3

  frontend:
    build:
      context: .
      dockerfile: packages/frontend/Dockerfile
      args:
        VITE_SIP_WSS_URL: ${VITE_SIP_WSS_URL:-}
        VITE_SIP_DOMAIN: ${VITE_SIP_DOMAIN:-}
        VITE_SIP_PASSWORD: ${VITE_SIP_PASSWORD:-}
    container_name: crm-frontend
    restart: unless-stopped
    ports:
      - '${APP_PORT:-80}:80'
      - '443:443'
    volumes:
      - ./ssl:/etc/nginx/ssl:ro
    depends_on:
      - backend
    networks:
      - external

volumes:
  pgdata:
  redisdata:

networks:
  internal:
    driver: bridge
  external:
    driver: bridge
[exit=0]
